Ch2-2 Rendering Loop¶

这节课中我们将构建一个最基本的渲染循环。

创建Semaphore和Fence¶

在主函数中， easyVulkan::CreateRpwf_Screen(); 后面加入以下内容：
using namespace vulkan;

semaphore semaphore_imageIsAvailable;
semaphore semaphore_renderingIsOver;
fence fence(true);
这里使用了之前自建的semaphore类，其默认构造器会创建一个默认的布尔semaphore。
这里的fence我创建了一个已经点亮了的，原因是我打算在Rendering Loop的开头进行等待，如果fence没被点亮，那么将永远无法进入第一次渲染。
你也可以调整顺序，将等待fence的代码放在每次循环的最后，但是届时实现即时帧的话会更麻烦。
两个semaphore，用处都写在名称上了，一个用于在获取图像后被点亮，一个用于渲染结束后被点亮。
fence用于命令缓冲区执行结束后被点亮。

获取下一张图像¶

在 graphicsBase 中加入以下内容：

    uint32_t currentImageIndex = 0;
public:
    uint32_t CurrentImageIndex() const { return currentImageIndex; }

    VkResult SwapImage(VkSemaphore semaphore_imageIsAvailable) {
        switch (VkResult result = vkAcquireNextImageKHR(device, swapchain, UINT64_MAX, semaphore_imageIsAvailable, VK_NULL_HANDLE, &currentImageIndex))
        {
            case VK_SUBOPTIMAL_KHR:
            case VK_ERROR_OUT_OF_DATE_KHR:
                RecreateSwapchain();
                [[fallthrough]];
            case VK_SUCCESS:
                return VK_SUCCESS;
            default:
                std::cout << "[ graphicsBase ]\nFailed to acquire the next image!\nError code: " << std::hex << result << std::endl;
                return result;
        }
    }
VkResult VKAPI_CALL vkAcquireNextImageKHR(
VkDevice device,
VkSwapchainKHR swapchain,
uint64_t timeout, //超时时间
VkSemaphore semaphore, //成功获取图像后点亮的semaphore
VkFence fence, //成功获取图像后点亮的fence
uint32_t * pImageIndex //用于保存所取得的图像的索引
);
超时时间用 UINT64_MAX 表示获取不到的话就一直等待下去。
semaphore 和 fence 至少要填一个（否则验证层会报错），一般使用semaphore，并且只用semaphore就够了。
在 vkAcquireNextImageKHR 所可能返回的所有错误代码中，VK_SUBOPTIMAL_KHR 或 VK_ERROR_OUT_OF_DATE_KHR 表示交换链需要被重建（主要由于窗口大小变换）。

创建命令缓冲区¶

首先我们要创建命令池，在我写的 commandPool 类中提供了两个用于创建命令池的函数（以及跟两个函数参数一致的构造器）：
VkResult Create(VkCommandPoolCreateInfo& createInfo) {
    createInfo.sType = VK_STRUCTURE_TYPE_COMMAND_POOL_CREATE_INFO;
    VkResult result = vkCreateCommandPool(graphicsBase::Base().Device(), &createInfo, nullptr, &handle);
    if (result)
        std::cout << "[ commandPool ]\nFailed to create a command pool!\nError code: " << std::hex << result << std::endl;
    return result;
}
VkResult Create(VkCommandPoolCreateFlags createFlags, uint32_t queueFamilyIndex) {
    VkCommandPoolCreateInfo createInfo{};
    createInfo.flags = createFlags;
    createInfo.queueFamilyIndex = queueFamilyIndex;
    return Create(createInfo);
}
我会用跟第二个函数参数一致的构造器来创建命令池，一般而言，你只需要flags和队列族索引就够了。
将以下代码放在主函数中，创建同步对象的代码之后：
//Create Command Objects
commandPool commandPool(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Graphics());
VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT表示我们可以重置所分配的命令缓冲区，那么我们可以反复录制并执行同一个命令缓冲区。
commandPool 类中提供了分配命令缓冲区的的函数：
VkResult AllocateCommandBuffers(VkCommandBuffer* pCommandBuffers, size_t commandBufferCount = 1, VkCommandBufferLevel level = VK_COMMAND_BUFFER_LEVEL_PRIMARY) const {
    VkCommandBufferAllocateInfo commandBufferAllocateInfo{};
    commandBufferAllocateInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
    commandBufferAllocateInfo.commandPool = handle;
    commandBufferAllocateInfo.level = level;
    commandBufferAllocateInfo.commandBufferCount = commandBufferCount;
    VkResult result = vkAllocateCommandBuffers(graphicsBase::Base().Device(), &commandBufferAllocateInfo, pCommandBuffers);
    if (result)
        std::cout << "[ commandPool ]\nFailed to allocate command buffers!\nError code: " << std::hex << result << std::endl;
}
你只需关注参数，而参数中唯一需要说明的是level，不过这个我们也已经在Ch1-4说明过了。
我将默认参数设置为VK_COMMAND_BUFFER_LEVEL_PRIMARY，你可以不使用二级命令缓冲区，但必定会使用一级命令缓冲区，因为只有一级才可以被直接提交。
于是延续之前创建命令池的代码，来创建一个一级缓冲区：
//Create Command Objects
commandPool commandPool(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Graphics());
commandBuffer commandBuffer_graphics;
commandPool.AllocateCommandBuffers((VkCommandBuffer*)&commandBuffer_graphics);
这里的命令缓冲区叫 commandBuffer_graphics 是因为以后还会有用于transfer以及在用于呈现的队列族下执行的命令缓冲区。

录制命令缓冲区¶

开始和结束命令缓冲区¶

commandPool 类中勇于用于开始和结束命令缓冲区的代码：
VkResult Begin(VkCommandBufferUsageFlags usageFlags = 0, const VkCommandBufferInheritanceInfo& inheritanceInfo = *(VkCommandBufferInheritanceInfo*)nullptr) const {
    VkCommandBufferBeginInfo beginInfo{};
    beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
    beginInfo.flags = usageFlags;
    beginInfo.pInheritanceInfo = &inheritanceInfo;
    VkResult result = vkBeginCommandBuffer(handle, &beginInfo);
    if (result)
        std::cout << "[ commandBuffer ]\nFailed to begin a command buffer!\nError code: " << std::hex << result << std::endl;
    return result;
}
VkResult End() const {
    VkResult result = vkEndCommandBuffer(handle);
    if (result)
        std::cout << "[ commandBuffer ]\nFailed to end a command buffer!\nError code: " << std::hex << result << std::endl;
    return result;
}
惯例你只需要关注参数。
usageFlags中可用的bit有：
VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT //表示该命令缓冲区只提交一次，然后执行完后就会被重置或Free掉
VK_COMMAND_BUFFER_USAGE_RENDER_PASS_CONTINUE_BIT //用于二级命令缓冲区，表示该命令缓冲区中的命令被完全包含在一个渲染通道之内
VK_COMMAND_BUFFER_USAGE_SIMULTANEOUS_USE_BIT //表示该命令缓冲区在待决（pending）状态下可以被重新提交（对于一级）或被录入一级缓冲区（对于二级）。
命令缓冲区在开始录制时会被重置，录制完成后进入可执行状态，然后在被提交（对于一级）或被录入一级缓冲区（对于二级）后会进入待决状态。
若开始录制时没有注明ONE_TIME_SUBMIT，执行完以后重归可执行状态，否则进入不可用（invalid）状态，直到被重置后重新录制为止。
inheritanceInfo用于二级命令缓冲区，暂不做讲解。

开始和结束渲染通道¶

为了偷懒不用手动创建一个 VkRenderPassBeginInfo ，我在 renderPass 中写了好几个用于开始命令缓冲区的函数。
不过我们只需关注最基础的一个就行了：
void CmdBegin(VkCommandBuffer commandBuffer, VkRenderPassBeginInfo& beginInfo, VkSubpassContents subpassContents = VK_SUBPASS_CONTENTS_INLINE) const {
    beginInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
    beginInfo.renderPass = handle;
    vkCmdBeginRenderPass(commandBuffer, &beginInfo, subpassContents);
}
void CmdEnd(VkCommandBuffer commandBuffer) const {
    vkCmdEndRenderPass(commandBuffer);
}
struct VkRenderPassBeginInfo {
VkStructureType sType;
const void * pNext;
VkRenderPass renderPass;
VkFramebuffer framebuffer;
VkRect2D renderArea;
uint32_t clearValueCount;
const VkClearValue * pClearValues;
};
如果我们之前在创建渲染通道时，在附件描述中指定了读取附件时的操作是清除，那么就得相应提供清空值。
清空值是 VkClearClorValue 或 VkDepthStencilClearValue 类型，根据你的附件类型选择相应类型。
如果你有些附件需要被清空，有些不要，那也得传入与 VkAttachmentDescription 数组对应数量的清空值，那些不需要被清空的附件会直接无视对应索引的清空值。
VkSubpassContents 可选：
VK_SUBPASS_CONTENTS_INLINE //表示命令会被内联进一级缓冲区，此时不准执行二级缓冲区
VK_SUBPASS_CONTENTS_SECONDARY_COMMAND_BUFFERS //表示会在一级命令缓冲区中执行二级缓冲区，且执行二级缓冲区将是子通道中唯一可用的命令
这两个可选项都是用于一级缓冲区的。
你不用计较Vulkan中的内联是如何实现的。如果一级缓冲区中不打算执行二级缓冲区，你肯定会使用VK_SUBPASS_CONTENTS_INLINE，因为你必须填入一个合法的选项，以及VK_SUBPASS_CONTENTS_INLINE的数值为0，换言之可以算是默认项。

提交命令缓冲区¶

在 graphicsBase 中加入以下公有函数：

VkResult SubmitGraphicsCommandBuffer(VkSubmitInfo& submitInfo, VkFence fence = VK_NULL_HANDLE) {
    submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
    VkResult result = vkQueueSubmit(queue_graphics, 1, &submitInfo, fence);
    if (result)
        std::cout << "[ graphicsBase ]\nFailed to submit the graphics command buffer!\nError code: " << std::hex << result << std::endl;
    return result;
}
VkResult SubmitGraphicsCommandBuffer(VkCommandBuffer commandBuffer,
    VkSemaphore semaphore_imageIsAvailable = VK_NULL_HANDLE, VkSemaphore semaphore_renderingIsOver = VK_NULL_HANDLE, VkFence fence = VK_NULL_HANDLE) {
    static VkPipelineStageFlags waitDstStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
    VkSubmitInfo submitInfo{};
    if (semaphore_imageIsAvailable)
        submitInfo.waitSemaphoreCount = 1,
        submitInfo.pWaitSemaphores = &semaphore_imageIsAvailable,
        submitInfo.pWaitDstStageMask = &waitDstStage;
    submitInfo.commandBufferCount = 1;
    submitInfo.pCommandBuffers = &commandBuffer;
    if (semaphore_renderingIsOver)
        submitInfo.signalSemaphoreCount = 1,
        submitInfo.pSignalSemaphores = &semaphore_renderingIsOver;
    return SubmitGraphicsCommandBuffer(submitInfo, fence);
}
struct VkSubmitInfo {
VkStructureType sType;
const void * pNext;
uint32_t waitSemaphoreCount;
const VkSemaphore * pWaitSemaphores;
const VkPipelineStageFlags * pWaitDstStageMask;
uint32_t commandBufferCount;
const VkCommandBuffer * pCommandBuffers;
uint32_t signalSemaphoreCount;
const VkSemaphore * pSignalSemaphores;
};
waitSemaphore指所提交的命令缓冲区开始执行前（实际未必）需要等待的semaphore。
pWaitDstStageMask 中的元素说明在哪个阶段等待对应索引的semaphore。
signalSemaphore指所提交的命令缓冲区全部执行完后点亮的semaphore。
对于图形命令，我们等待的通常会是VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT阶段。

呈现图像¶

在 graphicsBase 中加入以下公有函数：

VkResult PresentImage(VkPresentInfoKHR& presentInfo) {
    presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
    switch (VkResult result = vkQueuePresentKHR(queue_presentation, &presentInfo))
    {
        case VK_SUBOPTIMAL_KHR:
        case VK_ERROR_OUT_OF_DATE_KHR:
            RecreateSwapchain();
            [[fallthrough]];
        case VK_SUCCESS:
            return VK_SUCCESS;
        default:
            std::cout << "[ graphicsBase ]\nFailed to queue the image for presentation!\nError code: " << std::hex << result << std::endl;
            return result;
    }
}
VkResult PresentImage(VkSemaphore semaphore_renderingIsOver__Or_ownershipIsTransfered = VK_NULL_HANDLE) {
    VkPresentInfoKHR presentInfo{};
    if (semaphore_renderingIsOver__Or_ownershipIsTransfered)
        presentInfo.waitSemaphoreCount = 1,
        presentInfo.pWaitSemaphores = &semaphore_renderingIsOver__Or_ownershipIsTransfered;
    presentInfo.swapchainCount = 1;
    presentInfo.pSwapchains = &swapchain;
    presentInfo.pImageIndices = &currentImageIndex;
    return PresentImage(presentInfo);
}
跟获取图像一样，可能存在需要重建交换链的情况。
struct VkPresentInfoKHR {
VkStructureType sType;
const void * pNext;
uint32_t waitSemaphoreCount;
const VkSemaphore* pWaitSemaphores;
uint32_t swapchainCount;
const VkSwapchainKHR * pSwapchains;
const uint32_t * pImageIndices;
VkResult * pResults;
};
呈现时，等待的semaphore是没有所谓的对应阶段的，这些semaphore是在命令结束后被点亮的，这就意味着你得等命令完全执行结束后才能开始读取颜色附件。
你可以一次呈现多个交换链中的图像，虽然一般我们只会有一个交换链。
然后你需要制定各个交换链中需要呈现的图像的索引号。
最后，如果你呈现了多个交换链图像，若你想要接受各个结果，你得提供一个 VkResult 数组的地址。
由于我们只有一个交换链，所以我们会直接接受 vkQueuePresentKHR 返回的值。

构建Rendering Loop¶

为了使用我们在 EasyVulkan.hpp 中缩写的渲染通道，在 main.cpp 中include该文件。
然后直接看代码：
int main() {
    using namespace vulkan;
    if (!InitializeWindow({ 1280,720 }))
        return -1;
    easyVulkan::CreateRpwf_Screen();

    //Synchronization Object
    semaphore semaphore_imageIsAvailable;
    semaphore semaphore_renderingIsOver;
    fence fence(true);
    //Command Object
    commandPool commandPool(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Graphics());
    commandBuffer commandBuffer_graphics;
    commandPool.AllocateCommandBuffers((VkCommandBuffer*)&commandBuffer_graphics);

    //Clear Color
    VkClearValue clearColor = { .color = { 1.f, 0.f, 0.f, 1.f } };

    while (!glfwWindowShouldClose(pWindow)) {
        TitleFps();
        graphicsBase::Base().SwapImage(semaphore_imageIsAvailable);
        auto i = graphicsBase::Base().CurrentImageIndex();
        fence.WaitAndReset();

        commandBuffer_graphics.Begin(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
        easyVulkan::rpwf_screen.renderPass.CmdBegin(commandBuffer_graphics, easyVulkan::rpwf_screen.framebuffers[i], clearColor);


        easyVulkan::rpwf_screen.renderPass.CmdEnd(commandBuffer_graphics);
        commandBuffer_graphics.End();
        graphicsBase::Base().SubmitGraphicsCommandBuffer(commandBuffer_graphics, semaphore_imageIsAvailable, semaphore_renderingIsOver, fence);

        graphicsBase::Base().PresentImage(semaphore_renderingIsOver);
        glfwPollEvents();
    }
    TerminateWindow();
    return 0;
}
我们获取图像后，根据索引来选择相应的帧缓冲。
需要注意的是，
我打算在每个循环中重新录制命令缓冲区。
尤其是考虑到我比较喜欢用 vkCmdPushConstant 和 vkCmdUpdateBuffer 来更新并不很多的常量，这俩命令会把常量的数据直接记录在命令缓冲区里（惯例对不懂的内容先留个印象）。
如果你能确保你的命令缓冲区所记录的内容完在多次循环中完全一致，可以再循环前事先录制完命令缓冲区，注意这样的话不要使用VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT。
用于渲染的命令缓冲区结束后会同时点亮fence和semaphore。
因为我们在重新录制命令缓冲区前需要确保前一次的命令已完成执行，而你不能手动等待semaphore。
但是呈现图像时又只能等待semaphore，而不是fence。
你可以试着运行一下，你应该终于能看到一个全红的窗口了（因为我将 clearColor 写成了红色）。
这是用于绘制和呈现的队列族一致的情况下的代码，队列族所有权转移会在之后讲（其实没什么意义，既然我已近假定你能正常跑动以上代码了。不过届时会是一个演示Image Memory Barrier的机会）。

加快启动速度¶

这段会放在本节课讲，是因为直到显示出一个非全白的窗口为止，你无法意识到启动速度的快慢。
我们在已知平台和所用设备的情况下，可以通过跳过一些步骤来加快启动速度。
在 graphicsBase 中修改以下函数：
    void CheckInstanceLayersAndExtensions() {
#ifdef DISABLE_VK_LAYERS_AND_EXTENSIONS_CHECK
        return;
#endif
        //...
    }

    bool CheckDeviceLayersAndExtensions() {
#ifdef DISABLE_VK_LAYERS_AND_EXTENSIONS_CHECK
        return;
#endif
        //...
    }
接着就是你去EasyVKStart.h 中定义 DISABLE_VK_LAYERS_AND_EXTENSIONS_CHECK 了。
仅适用于你没有用一些不太通用的层和扩展的情况。（我们教程里所用的都是比较常见的层和扩展）
获取GLFW所需的扩展这一步比较耗时，我们可以事先知晓平台所需的扩展，直接push，以Windows为例：
（顺带跟你核对下整个 InitializeWindow ）
bool InitializeWindow(VkExtent2D windowSize, bool resizable = true, bool limit_framerate = true) {
    using namespace vulkan;
    if (!glfwInit())//glfwInit() returns 0 if FAIL
            return false;
    glfwWindowHint(GLFW_CLIENT_API, GLFW_NO_API);
    //Push extensions
#ifdef _WIN32
    graphicsBase::Base().PushInstanceExtension("VK_KHR_surface");
    graphicsBase::Base().PushInstanceExtension("VK_KHR_win32_surface");//就算你是64位系统也一样！
#else
    uint32_t extensionCount = 0;
    const char** extensionNames;
    extensionNames = glfwGetRequiredInstanceExtensions(&extensionCount);
    if (!extensionNames)
    {
            std::cout << "[ InitializeWindow ]\nVulkan is not available on this machine!\n";
            glfwTerminate();
            return false;
    }
    for (size_t i = 0; i < extensionCount; i++)
            graphicsBase::Base().PushInstanceExtension(extensionNames[i]);
#endif
    graphicsBase::Base().PushDeviceExtension(VK_KHR_SWAPCHAIN_EXTENSION_NAME);
    //Create a vulkan instance
    graphicsBase::Base().CreateInstance();
    //Get the monitor and the video mode (Not necessary, but maybe useful for some purposes.)
    pMonitor = glfwGetPrimaryMonitor();
    pMode = glfwGetVideoMode(pMonitor);
    //Create a glfw window
    glfwWindowHint(GLFW_RESIZABLE, resizable);
    pWindow = glfwCreateWindow(windowSize.width, windowSize.height, windowTitle, nullptr, nullptr);//Returns 0 if FAIL
    if (!pWindow)
    {
            std::cout << "[ InitializeWindow ]\nFailed to create a glfw window!\n";
            glfwTerminate();
            return false;
    }
    //Create a vulkan surface
    VkSurfaceKHR surface = VK_NULL_HANDLE;
    if (VkResult result = glfwCreateWindowSurface(graphicsBase::Base().Instance(), pWindow, nullptr, &surface))
    {
            std::cout << "[ InitializeWindow ]\nFailed to create a window surface!\nError code: " << std::hex << result << std::endl;
            glfwTerminate();
            return false;
    }
    graphicsBase::Base().Surface(surface);
    //Get a physical device
    graphicsBase::Base().GetPhysicalDevice();
    //Create a logical device and a swapchain
    graphicsBase::Base().CreateLogicalDevice();
    graphicsBase::Base().CreateSwapchain(limit_framerate);
    return true;
}
