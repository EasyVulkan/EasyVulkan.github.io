Ch2-4 即时帧与队列族所有权转移¶

即时帧¶

我们之前的Rendering Loop是构建在一个命令缓冲区和一套同步对象上的，我们在每一帧里等待上一帧的fence被点亮，然而，若此时上一帧的命令还没有执行完，那么等待fence就需要花时间。
实现即时帧就是创建多个命令缓冲区和多套同步对象（以及多套uniform buffer、描述符、以及渲染附件），并交替使用它们。这样一来，若有N套即时帧，当前帧等待的fence就会对应N帧前的命令，相比没有即时帧，这些命令很有可能已经被执行完了，于是就可以减少阻塞。
比较推荐的做法是为每张交换链图像实现一套即时帧 —— 前提是你的程序不是特别消耗内存的话，如果你要在4K下实现4xMSAA延迟渲染并使用即时帧，除非你是特别强劲的显卡，不然显存八成是会炸的。
int main() {
    using namespace vulkan;
    if (!InitializeWindow({ 1280, 720 }))
        return -1;
    easyVulkan::CreateRpwf_Screen();
    CreateLayout();
    CreatePipeline();

    //Synchronization Object
    uint32_t framesInFlightCount = std::min(uint32_t(graphicsBase::Base().SwapchainImageCount()), 3u);//使用3套即时帧，若交换链图像数量少于3，则使用交换链图像数量
    std::vector<semaphore> semaphores_imageIsAvailable(framesInFlightCount);
    std::vector<semaphore> semaphores_renderingIsOver(framesInFlightCount);
    std::vector<fence> fences;
    for (size_t i = 0; i < framesInFlightCount; i++)
        fences.emplace_back(true);
    //Command Object
    commandPool commandPool(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Graphics());
    std::vector<commandBuffer> commandBuffers_graphics(framesInFlightCount);
    commandPool.AllocateCommandBuffers((VkCommandBuffer*)commandBuffers_graphics.data(), framesInFlightCount);

    //Clear Color
    VkClearValue clearColor = { .color = { 1.f, 0.f, 0.f, 1.f } };

    while (!glfwWindowShouldClose(pWindow)) {
        TitleFps();
        static uint32_t i = 0;
        graphicsBase::Base().SwapImage(semaphores_imageIsAvailable[i]);
        fences[i].WaitAndReset();

        commandBuffers_graphics[i].Begin(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
        //注意，framebuffer的编号是graphicsBase::Base().CurrentImageIndex()，而不是i。
        easyVulkan::rpwf_screen.renderPass.CmdBegin(commandBuffers_graphics[i], easyVulkan::rpwf_screen.framebuffers[graphicsBase::Base().CurrentImageIndex()], clearColor) ;

        vkCmdBindPipeline(commandBuffers_graphics[i], VK_PIPELINE_BIND_POINT_GRAPHICS, pipeline_triangle);
        vkCmdDraw(commandBuffers_graphics[i], 3, 1, 0, 0);

        easyVulkan::rpwf_screen.renderPass.CmdEnd(commandBuffers_graphics[i]);
        commandBuffers_graphics[i].End();
        graphicsBase::Base().SubmitGraphicsCommandBuffer(commandBuffers_graphics[i], semaphores_imageIsAvailable[i], semaphores_renderingIsOver[i], fences[i]);

        graphicsBase::Base().PresentImage(semaphores_renderingIsOver[i]);
        i = (i + 1) % framesInFlightCount;

        glfwPollEvents();
    }
    TerminateWindow();
    return 0;
}
需要注意的是，即时帧的数量不应该多于交换链图像的数量。举例而言，如果有3个交换链图像，而即时帧数量为4，那么当第一次 WaitAndReset 第4套即时帧的fence时，由于我们是以signal状态创建的fence，实质上就不会发生等待，但第1套即时帧跟第4套却会使用同一个交换链图像，而第1套即时帧的命令却可能还没有执行完！

队列族所有权转移¶

资源在被队列使用时，其队列族拥有其所有权，当你把资源从一个队列族中的队列，移交给另一个队列族中的队列时，需要进行队列族所有权转移。
举例而言，若进行图形绘制的队列和进行呈现的队列不在一个队列族中，你可能就需要进行队列所有权转移（如我之前在Ch1-2所说，至少我的Nvidia 1060MQ不需要，我很确信我手动选择了两个不同的队列族）。
队列所有权转移通过两个可以是完全一致的image memory barrier，在不同队列中执行，来实行。两个一组，一个释放所有权，一个取得所有权。如果有其他的同步措施，你可以在第二个image memory barrier中少写一些参数。
在 graphicsBase 中加入下列代码：
//第一个barrier，必须在graphics的queue中，在渲染通道结束后执行。
void CmdReleaseImageOwnership_Graphics(VkCommandBuffer commandBuffer) {
    VkImageMemoryBarrier imageMemoryBarrier_g2p_release{};
    imageMemoryBarrier_g2p_release.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
    imageMemoryBarrier_g2p_release.srcAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
    imageMemoryBarrier_g2p_release.dstAccessMask;
    imageMemoryBarrier_g2p_release.oldLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
    imageMemoryBarrier_g2p_release.newLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
    imageMemoryBarrier_g2p_release.srcQueueFamilyIndex = queueFamilyIndex_graphics;
    imageMemoryBarrier_g2p_release.dstQueueFamilyIndex = queueFamilyIndex_presentation;
    imageMemoryBarrier_g2p_release.image = swapchainImages[currentImageIndex];
    imageMemoryBarrier_g2p_release.subresourceRange = { VK_IMAGE_ASPECT_COLOR_BIT, 0, 1, 0, 1 };
    vkCmdPipelineBarrier(commandBuffer, VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT, VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT, 0,
        0, nullptr, 0, nullptr, 1, &imageMemoryBarrier_g2p_release);
}
//第二个barrier，必须在presentation的queue中执行。
void CmdAcquireImageOwnership_Presentation(VkCommandBuffer commandBuffer) {
    VkImageMemoryBarrier imageMemoryBarrier_g2p_acquire{};
    imageMemoryBarrier_g2p_acquire.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
    imageMemoryBarrier_g2p_acquire.oldLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
    imageMemoryBarrier_g2p_acquire.newLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
    imageMemoryBarrier_g2p_acquire.srcQueueFamilyIndex = queueFamilyIndex_graphics;
    imageMemoryBarrier_g2p_acquire.dstQueueFamilyIndex = queueFamilyIndex_presentation;
    imageMemoryBarrier_g2p_acquire.image = swapchainImages[currentImageIndex];
    imageMemoryBarrier_g2p_acquire.subresourceRange = { VK_IMAGE_ASPECT_COLOR_BIT, 0, 1, 0, 1 };
    vkCmdPipelineBarrier(commandBuffer, VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT, VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT, 0,
        0, nullptr, 0, nullptr, 1, &imageMemoryBarrier_g2p_acquire);
}
//向presentation队列提交命令缓冲区
VkResult SubmitPresentationCommandBuffer(VkCommandBuffer commandBuffer,
    VkSemaphore semaphore_renderingIsOver = VK_NULL_HANDLE, VkSemaphore semaphore_ownershipIsTransfered = VK_NULL_HANDLE, VkFence fence = VK_NULL_HANDLE) {
    static VkPipelineStageFlags waitDstStage = VK_PIPELINE_STAGE_ALL_COMMANDS_BIT;
    VkSubmitInfo submitInfo{};
    submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
    if (semaphore_renderingIsOver)
        submitInfo.waitSemaphoreCount = 1,
        submitInfo.pWaitSemaphores = &semaphore_renderingIsOver,
        submitInfo.pWaitDstStageMask = &waitDstStage;
    submitInfo.commandBufferCount = 1;
    submitInfo.pCommandBuffers = &commandBuffer;
    if (semaphore_ownershipIsTransfered)
        submitInfo.signalSemaphoreCount = 1,
        submitInfo.pSignalSemaphores = &semaphore_ownershipIsTransfered;
    VkResult result = vkQueueSubmit(queue_presentation, 1, &submitInfo, fence);
    if (result)
        std::cout << "[ graphicsBase ]\nFailed to submit the presentation command buffer!\nError code: " << std::hex << result << std::endl;
    return result;
}
首先来解释下 vkCmdPipelineBarrier ：
void VKAPI_CALL vkCmdPipelineBarrier(
VkCommandBuffer commandBuffer,
VkPipelineStageFlags srcStageMask,
VkPipelineStageFlags dstStageMask,
VkDependencyFlags dependencyFlags,
uint32_t memoryBarrierCount,
const VkMemoryBarrier * pMemoryBarriers,
uint32_t bufferMemoryBarrierCount,
const VkBufferMemoryBarrier * pBufferMemoryBarriers,
uint32_t imageMemoryBarrierCount,
const VkImageMemoryBarrier * pImageMemoryBarriers
);
相信这些参数你都看得懂。类似于之前讲过的子通道依赖， vkCmdPipelineBarrier 会使得 srcStageMask 中的操作在 dstStageMask 中的操作之前发生，并在两者之间实行Memory Barrier。
struct VkImageMemoryBarrier {
VkStructureType sType;
const void * pNext;
VkAccessFlags srcAccessMask;
VkAccessFlags dstAccessMask;
VkImageLayout oldLayout;
VkImageLayout newLayout;
uint32_t srcQueueFamilyIndex;
uint32_t dstQueueFamilyIndex;
VkImage image;
VkImageSubresourceRange subresourceRange;
};
srcAccessMask 和 dstAccessMask 想必不用多讲了。
oldLayout 和 newLayout 指定了Memory Barrier前后的图像布局，如果你不需要图像布局转换，那么这里就填两个一样的值。
srcQueueFamilyIndex 和 dstQueueFamilyIndex 指定了Memory Barrier前后，该资源属于哪个队列族，若不需要转移所有权，可以写两个VK_QUEUE_FAMILY_IGNORED。
VkImageSubresourceRange 在Ch1-3中解释过了。
于是第一个pipeline barrier，从graphics队列族释放所有权：
VkImageMemoryBarrier imageMemoryBarrier_g2p_release{};
imageMemoryBarrier_g2p_release.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
imageMemoryBarrier_g2p_release.srcAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
imageMemoryBarrier_g2p_release.dstAccessMask;//若对应的阶段是TOP/BOTTOM_OF_PIPE，你可以使用VK_ACCESS_MEMORY_LOAD_BIT和VK_ACCESS_MEMORY_WRITE_BIT，但是没有必要。
imageMemoryBarrier_g2p_release.oldLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;//渲染通道结束时自动转换成了VK_IMAGE_LAYOUT_PRESENT_SRC_KHR
imageMemoryBarrier_g2p_release.newLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;//不做图像布局变换
imageMemoryBarrier_g2p_release.srcQueueFamilyIndex = queueFamilyIndex_graphics;//执行图形命令的队列族
imageMemoryBarrier_g2p_release.dstQueueFamilyIndex = queueFamilyIndex_presentation;//用于呈现的队列族
imageMemoryBarrier_g2p_release.image = swapchainImages[currentImageIndex];//当前交换链图像
imageMemoryBarrier_g2p_release.subresourceRange = { VK_IMAGE_ASPECT_COLOR_BIT, 0, 1, 0, 1 };
//dstStageMask是VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT，意味着不进行任何阻塞。
//这个Barrier之后不该有其他与该交换链图像相关的命令了，所以不进行阻塞毫无问题，我们只需在颜色被写入到图像后释放所有权。
//dependencyFlags使用了0，可以用BY_REGION但没必要，毕竟，显卡驱动不至于给每个像素或采样点标记所有权。
vkCmdPipelineBarrier(commandBuffer, VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT, VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT, 0,
    0, nullptr, 0, nullptr, 1, &imageMemoryBarrier_g2p_release);
第二个pipeline barrier，在presentation队列族获取所有权：
//这个barrier可以跟上面的完全一致，我有意将其写得不一致，以说明哪些地方可以做出区别。
VkImageMemoryBarrier imageMemoryBarrier_g2p_acquire{};
imageMemoryBarrier_g2p_acquire.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
//因为有Semaphore做同步，确保在执行这个Barrier所在的命令缓冲区前完成COLOR_ATTACHMENT_WRITE，所以srcAccessMask可以不写
imageMemoryBarrier_g2p_acquire.oldLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
imageMemoryBarrier_g2p_acquire.newLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
imageMemoryBarrier_g2p_acquire.srcQueueFamilyIndex = queueFamilyIndex_graphics;
imageMemoryBarrier_g2p_acquire.dstQueueFamilyIndex = queueFamilyIndex_presentation;
imageMemoryBarrier_g2p_acquire.image = swapchainImages[currentImageIndex];
imageMemoryBarrier_g2p_acquire.subresourceRange = { VK_IMAGE_ASPECT_COLOR_BIT, 0, 1, 0, 1 };
//也是因为有Semaphore做同步，srcStage我可以使用VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT。
vkCmdPipelineBarrier(commandBuffer, VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT, VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT, 0,
    0, nullptr, 0, nullptr, 1, &imageMemoryBarrier_g2p_acquire);
你将包含以上barrier的命令缓冲区提交到presentation队列的方式跟之前所写的 SubmitGraphicsCommandBuffer 并无太大不同，只是 waitDstStage 变成了 VK_PIPELINE_STAGE_ALL_COMMANDS_BIT，渲染通道结束时的隐式子通道依赖中，srcDstStage是VK_PIPELINE_STAGE_ALL_COMMANDS_BIT，这里与其对应。
最后主函数变成了这样：
int main() {
    using namespace vulkan;
    if (!InitializeWindow({ 1280, 720 }))
        return -1;
    easyVulkan::BootScreen();
    CreateLayout();
    CreatePipeline();

    //Synchronization object
    semaphore semaphore_imageIsAvailable;
    semaphore semaphore_renderingIsOver;
    semaphore semaphore_ownershipIsTransfered;
    fence fence_graphics(true);
    fence fence_presentation(true);
    //Create Command Objects
    commandPool commandPool_graphics(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Graphics());
    commandBuffer commandBuffer_graphics;
    commandPool_graphics.AllocateCommandBuffers((VkCommandBuffer*)&commandBuffer_graphics);

    commandPool commandPool_presentation(VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, graphicsBase::Base().QueueFamilyIndex_Presentation());
    commandBuffer commandBuffer_presentation;
    commandPool_presentation.AllocateCommandBuffers((VkCommandBuffer*)&commandBuffer_presentation);

    //Clear Color
    VkClearValue clearColor = { .color = { 1.f, 0.f, 0.f, 1.f } };

    while (!glfwWindowShouldClose(pWindow)) {
        TitleFps();
        graphicsBase::Base().SwapImage(semaphore_imageIsAvailable);
        auto i = graphicsBase::Base().CurrentImageIndex();

        //Graphics Queue Side
        fence_graphics.WaitAndReset();
        commandBuffer_graphics.Begin(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
        easyVulkan::rpwf_screen.renderPass.CmdBegin(commandBuffer_graphics, easyVulkan::rpwf_screen.framebuffers[i], clearColor);

        vkCmdBindPipeline(commandBuffer_graphics, VK_PIPELINE_BIND_POINT_GRAPHICS, pipeline_triangle);
        vkCmdDraw(commandBuffer_graphics, 3, 1, 0, 0);

        easyVulkan::rpwf_screen.renderPass.CmdEnd(commandBuffer_graphics);
        graphicsBase::Base().CmdReleaseImageOwnership_Graphics(commandBuffer_graphics);
        commandBuffer_graphics.End();
        graphicsBase::Base().SubmitGraphicsCommandBuffer(commandBuffer_graphics, semaphore_imageIsAvailable, semaphore_renderingIsOver, fence_graphics);

        //Presentation Queue Side
        fence_presentation.WaitAndReset();
        commandBuffer_presentation.Begin(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
        graphicsBase::Base().CmdAcquireImageOwnership_Presentation(commandBuffer_presentation);
        commandBuffer_presentation.End();
        graphicsBase::Base().SubmitPresentationCommandBuffer(commandBuffer_presentation, semaphore_renderingIsOver, semaphore_ownershipIsTransfered, fence_presentation);

        graphicsBase::Base().PresentImage(semaphore_ownershipIsTransfered);
        glfwPollEvents();
    }
    TerminateWindow();
    return 0;
}

因为多了一个命令缓冲区，所以相应地也多了个fence和semaphore。
